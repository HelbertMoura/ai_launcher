# v5.5.0 "Terminal Dramático" Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Transformar o AI Launcher Pro em um dev tool com identidade visual forte (Terminal Dramático) — split do App.tsx monolítico, design system expandido, dark+light reais, iconografia unificada (lucide), typography JetBrains Mono self-host, redesign das 9 superfícies, motion coreografada e 4 extras (update checker, onboarding polido, atalhos visíveis, export/import config).

**Architecture:** Tauri v2 (Rust) + React 19 (upgrade de 18) + Vite 8 (upgrade de 5) + TypeScript. Split do `App.tsx` (1744 linhas) em `src/layout/`, `src/tabs/`, `src/shared/`, `src/icons/`. Design tokens em `src/styles/tokens-dark.css` + `tokens-light.css` + `fonts.css` auto-carregados via `main.tsx`. Lucide-react para ícones, PresetIcon existente preservado.

**Tech Stack:** React 19, Vite 8, TypeScript 5, Tauri v2.8+, lucide-react, JetBrains Mono (self-host woff2), ureq (Rust HTTP client já instalado).

**Spec:** `docs/superpowers/specs/2026-04-20-v5.5-terminal-dramatico-design.md`

```tsx
interface SparklineProps {
  points: number[];
  width?: number;
  height?: number;
  stroke?: string;
}

export function Sparkline({ points, width = 80, height = 24, stroke = 'currentColor' }: SparklineProps) {
  if (points.length === 0) return null;
  const max = Math.max(...points, 0.01);
  const min = Math.min(...points, 0);
  const range = max - min || 1;
  const stepX = width / Math.max(points.length - 1, 1);
  const d = points.map((p, i) => {
    const x = i * stepX;
    const y = height - ((p - min) / range) * height;
    return `${i === 0 ? 'M' : 'L'}${x.toFixed(1)},${y.toFixed(1)}`;
  }).join(' ');
  return (
    <svg width={width} height={height} viewBox={`0 0 ${width} ${height}`} aria-hidden="true">
      <path d={d} fill="none" stroke={stroke} strokeWidth={1.5} strokeLinecap="round" strokeLinejoin="round" />
    </svg>
  );
}
```

```tsx
import { useEffect, useState } from 'react';
import { invoke } from '@tauri-apps/api/core';
import { ArrowUpRight } from '../icons';

const CACHE_KEY = 'ai-launcher:latest-release-check';
const CACHE_TTL_MS = 6 * 60 * 60 * 1000;

interface LatestRelease { tag_name: string; html_url: string; }

function isNewer(current: string, latest: string): boolean {
  const c = current.replace(/^v/, '').split('.').map(Number);
  const l = latest.replace(/^v/, '').split('.').map(Number);
  for (let i = 0; i < 3; i++) {
    if ((l[i] ?? 0) > (c[i] ?? 0)) return true;
    if ((l[i] ?? 0) < (c[i] ?? 0)) return false;
  }
  return false;
}

export function StatusBar({ version, provider, activeTab }: { version: string; provider: string; activeTab: string }) {
  const [latest, setLatest] = useState<LatestRelease | null>(null);

  useEffect(() => {
    const cached = localStorage.getItem(CACHE_KEY);
    if (cached) {
      try {
        const { ts, data } = JSON.parse(cached);
        if (Date.now() - ts < CACHE_TTL_MS) {
          setLatest(data);
          return;
        }
      } catch {}
    }
    invoke<LatestRelease>('check_latest_release')
      .then(data => {
        setLatest(data);
        localStorage.setItem(CACHE_KEY, JSON.stringify({ ts: Date.now(), data }));
      })
      .catch(() => {});
  }, []);

  const updateAvailable = latest && isNewer(version, latest.tag_name);

  return (
    <footer className="statusbar">
      <span>[ai-launcher {version}]</span>
      <span>●  {provider}</span>
      <span>tab: {activeTab}</span>
      <span className="statusbar__spacer" />
      {updateAvailable && latest && (
        <a
          href="#"
          className="statusbar__update"
          onClick={(e) => {
            e.preventDefault();
            // Abrir no browser via tauri shell
            import('@tauri-apps/plugin-shell').then(({ open }) => open(latest.html_url));
          }}
        >
          <ArrowUpRight size={12} /> {latest.tag_name} available
        </a>
      )}
    </footer>
  );
}
```

```typescript
interface ConfigDump {
  version: string;
  exportedAt: string;
  providers: Array<Record<string, unknown>>;
  presets: Array<Record<string, unknown>>;
  settings: Record<string, unknown>;
}

export function exportConfig(appVersion: string): string {
  const providers = JSON.parse(localStorage.getItem('ai-launcher:providers') ?? '[]');
  const presets = JSON.parse(localStorage.getItem('ai-launcher:presets') ?? '[]');
  const settings = {
    theme: localStorage.getItem('ai-launcher:theme'),
    displayFont: localStorage.getItem('ai-launcher:display-font'),
    hideWelcome: localStorage.getItem('ai-launcher:hide-welcome'),
  };

  // Redact sensitive fields in providers
  const redacted = providers.map((p: Record<string, unknown>) => {
    const env = { ...(p.env as Record<string, string>) };
    for (const k of Object.keys(env)) {
      if (/token|key|secret|bearer/i.test(k)) {
        env[k] = '{{REDACTED}}';
      }
    }
    return { ...p, env };
  });

  const dump: ConfigDump = {
    version: appVersion,
    exportedAt: new Date().toISOString(),
    providers: redacted,
    presets,
    settings,
  };
  return JSON.stringify(dump, null, 2);
}

export function importConfig(raw: string, mode: 'merge' | 'replace'): { ok: true } | { ok: false; error: string } {
  let dump: ConfigDump;
  try {
    dump = JSON.parse(raw);
  } catch {
    return { ok: false, error: 'Arquivo não é JSON válido.' };
  }
  if (!dump.version || !Array.isArray(dump.providers)) {
    return { ok: false, error: 'Formato desconhecido.' };
  }

  if (mode === 'replace') {
    localStorage.setItem('ai-launcher:providers', JSON.stringify(dump.providers));
    localStorage.setItem('ai-launcher:presets', JSON.stringify(dump.presets));
  } else {
    const existingProviders = JSON.parse(localStorage.getItem('ai-launcher:providers') ?? '[]');
    const existingPresets = JSON.parse(localStorage.getItem('ai-launcher:presets') ?? '[]');
    localStorage.setItem('ai-launcher:providers', JSON.stringify([...existingProviders, ...dump.providers]));
    localStorage.setItem('ai-launcher:presets', JSON.stringify([...existingPresets, ...dump.presets]));
  }

  const s = dump.settings ?? {};
  if (typeof s.theme === 'string') localStorage.setItem('ai-launcher:theme', s.theme);
  if (typeof s.displayFont === 'string') localStorage.setItem('ai-launcher:display-font', s.displayFont);

  return { ok: true };
}
```
