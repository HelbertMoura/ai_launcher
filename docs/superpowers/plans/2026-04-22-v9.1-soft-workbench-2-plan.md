- [ ] **Step 1: Update appearance.ts — add accent color application from CSS**

```typescript
// Add after ACCENT_PRESETS:
export type AccentColor = string; // hex color

export function applyAccentColor(color: string): void {
  document.documentElement.style.setProperty('--accent', color);
  // Compute lighter variant for accent-light
  document.documentElement.style.setProperty('--accent-light', color + '1f'); // ~12% opacity
  document.documentElement.style.setProperty('--accent-hover', adjustBrightness(color, -15));
}

// Helper: darken/lighten a hex color
function adjustBrightness(hex: string, percent: number): string {
  const num = parseInt(hex.replace('#', ''), 16);
  const r = Math.min(255, Math.max(0, (num >> 16) + percent));
  const g = Math.min(255, Math.max(0, ((num >> 8) & 0x00FF) + percent));
  const b = Math.min(255, Math.max(0, (num & 0x0000FF) + percent));
  return `#${((r << 16) | (g << 8) | b).toString(16).padStart(6, '0')}`;
}
```
